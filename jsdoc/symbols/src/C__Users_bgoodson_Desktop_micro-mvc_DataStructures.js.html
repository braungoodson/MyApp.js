 16 /**
 17 * Common Dump Function : A simple debugging method that dumps the contents of 
 18 *  an array into the console.
 19 *
 20 * @method dump
 21 * @returns {void} Doesn't return anything.
 22 */
 23 DataStructures.Common.dump = function () {
 24 	var out = '';
 25 	var l = this.length;
 26 	console.log("* Dumping " + this.id + " of length: " + l);
 27 	while (l--){
 28 		out += this[l].toString() + ';';
 29 	}
 30 	console.log("* Dump: " + out);
 31 }

 60 /**
 61 * LinkedList Data Structure : Contains basic Stack and LinkedList.
 62 *
 63 * @namespace DataStructures
 64 * @class LinkedList
 65 * @constructor
 66 */
 67 DataStructures.LinkedList = function () {
 68 	/**
 69 	* Id of the list.
 70 	* @property id
 71 	* @type String
 72 	*/
 73 	this.id = "LinkedList";
 74 };
 75 DataStructures.LinkedList.prototype = new Array();
 76 /**
 77 * A simple debugging method that dumps the contents of an array
 78 *  into the console.
 79 *
 80 * @method dump
 81 * @function dump
 82 * @see DataStructures.Common.dump
 83 * @returns {void} Doesn't return anything.
 84 */
 85 DataStructures.LinkedList.prototype.dump = DataStructures.Common.dump;
 86 /**
 87 * Add an item to the list. Calls parent push method.
 88 *
 89 * @function add
 90 * @param {Object} item The item to add to the list.
 91 * @returns {void} Doesn't return anything.
 92 */
 93 DataStructures.LinkedList.prototype.add = function (item) {
 94 	this.push(item);
 95 }
 96 /**
 97 * Remove an item from the list. Calls parent splice method.
 98 *
 99 * @function remove
100 * @param {Object} item The item to remove from the list.
101 * @returns {void} Doesn't return anything.
102 */
103 DataStructures.LinkedList.prototype.remove = function (item) {
104 	this.splice(this.indexOf(item),1);
105 }
