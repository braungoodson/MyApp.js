 99 /**
100 * User's Table View : A table view. Essentially a data grid.
101 *
102 * @namespace MyApp.Views
103 * @class UsersTableView
104 * @constructor
105 * @see MVCSybSystem.AbstractView
106 * @param {AbstractModel} model This view's model.
107 * @returns {void}
108 */
109 MyApp.Views.UsersTableView = function (model) {
110 	this.construct(model);
111 	this.initialize();
112 	this.stage = document.createElement('div');
113 	this.stage.style.border = '1px solid black';
114 	this.datagrid = document.createElement('div');
115 	this.controls = document.createElement('div');
116 	this.editUserControl = document.createElement('input');
117 	this.editUserControl.type = 'submit';
118 	this.editUserControl.onclick = this.controller.onclick_eventHandler;
119 	this.editUserControl.innerHTML = 'edit user';
120 	this.controls.appendChild(this.editUserControl);
121 	this.stage.appendChild(this.datagrid);
122 	this.stage.appendChild(this.controls);
123 	document.body.appendChild(this.stage);
124 }
125 MyApp.Views.UsersTableView.prototype = new MVCSubSystem.AbstractView();
126 /**
127 * makeController : This view's method for making it's controller.
128 *
129 * @returns {MyApp.Controllers.TableController}
130 */
131 MyApp.Views.UsersTableView.prototype.makeController = function () {
132 	return new MyApp.Controllers.UsersTableController(this);
133 }
134 /**
135 * draw : This view's method for drawing itself based of it's model's data.
136 *
137 * @returns {void}
138 */
139 MyApp.Views.UsersTableView.prototype.draw = function () {
140 	this.datagrid.innerHTML = '';
141 	for (var i in this.model.users)
142 		if (typeof this.model.users[i] === "object")
143 			this.datagrid.innerHTML += 
144 				this.model.users[i].id + " | " +
145 				this.model.users[i].name + " | " +
146 				this.model.users[i].password +
147 				'<br />';
148 }

201 	return new MyApp.Controllers.UsersGraphController(this);
202 }
203 /**
204 * draw : This view's method for drawing itself based of it's model's data.
205 *
206 * @returns {void}
207 */
208 MyApp.Views.UsersGraphView.prototype.draw = function () {
209 	var numUsers = 0;
210 	this.graph.innerHTML = '';
211 	for (var i in this.model.users)
212 		if (typeof this.model.users[i] === "object") {
213 			numUsers++;
214 			this.graph.innerHTML += "<div style='height:10px;width:10px;background-color:red;float:left;border:1px solid black;margin:1px 1px 1px 1px;'></div>";
215 		}
216 		this.graph.innerHTML += " " + numUsers + " Users";
217 }
